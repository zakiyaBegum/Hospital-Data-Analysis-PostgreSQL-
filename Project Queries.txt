1️⃣ Total Number of Patients
SELECT SUM("Patients Count") AS total_patients
FROM hospital_data;
2️⃣ Average Number of Doctors per Hospital
SELECT AVG("Doctors Count") AS avg_doctors
FROM hospital_data;
3️⃣ Top 3 Departments with Highest Patients
SELECT "Department",
       SUM("Patients Count") AS total_patients
FROM hospital_data
GROUP BY "Department"
ORDER BY total_patients DESC
LIMIT 3;
4️⃣ Hospital with Maximum Medical Expenses
SELECT "Hospital Name",
       SUM("Medical Expenses") AS total_expenses
FROM hospital_data
GROUP BY "Hospital Name"
ORDER BY total_expenses DESC
LIMIT 1;
5️⃣ Daily Average Medical Expenses (per hospital)
SELECT "Hospital Name",
       TO_DATE("Admission Date", 'DD-MM-YYYY') AS admission_day,
       AVG("Medical Expenses") AS avg_daily_expense
FROM hospital_data
GROUP BY "Hospital Name", admission_day
ORDER BY "Hospital Name";
6️⃣ Longest Hospital Stay
SELECT *,
       (TO_DATE("Discharge Date", 'DD-MM-YYYY') -
        TO_DATE("Admission Date", 'DD-MM-YYYY')) AS stay_days
FROM hospital_data
ORDER BY stay_days DESC
LIMIT 1;
7️⃣ Total Patients Per City
SELECT "Location",
       SUM("Patients Count") AS total_patients
FROM hospital_data
GROUP BY "Location"
ORDER BY total_patients DESC;
8️⃣ Average Length of Stay Per Department
SELECT "Department",
       AVG(
           TO_DATE("Discharge Date", 'DD-MM-YYYY') -
           TO_DATE("Admission Date", 'DD-MM-YYYY')
       ) AS avg_stay_days
FROM hospital_data
GROUP BY "Department"
ORDER BY avg_stay_days DESC;
9️⃣ Department with Lowest Patients
SELECT "Department",
       SUM("Patients Count") AS total_patients
FROM hospital_data
GROUP BY "Department"
ORDER BY total_patients ASC
LIMIT 1;
🔟 Monthly Medical Expenses Report
SELECT DATE_TRUNC('month',
       TO_DATE("Admission Date", 'DD-MM-YYYY')) AS month,
       SUM("Medical Expenses") AS total_expenses
FROM hospital_data
GROUP BY month
ORDER BY month;
